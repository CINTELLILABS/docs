---
title: "Get Eval Agent"
api: "GET https://api.bland.ai/v1/evals/agents/{eval_agent_id}"
description: "Fetch one eval agent and its current draft version."
---

### Headers

<ParamField header="authorization" type="string" required>
  Your API key for authentication.
</ParamField>

### Path Parameters

<ParamField path="eval_agent_id" type="string" required>
  The unique identifier of the eval agent.
</ParamField>

### Response

<ResponseField name="data.agent" type="object">
  The requested eval agent.
</ResponseField>

<ResponseField name="data.agent.id" type="string">
  Unique identifier for the eval agent.
</ResponseField>

<ResponseField name="data.agent.org_id" type="string">
  ID of the organization that owns this eval agent.
</ResponseField>

<ResponseField name="data.agent.key" type="string">
  Unique slug key for the eval agent within the organization.
</ResponseField>

<ResponseField name="data.agent.name" type="string">
  Display name of the eval agent.
</ResponseField>

<ResponseField name="data.agent.description" type="string | null">
  Optional description of what the eval agent grades.
</ResponseField>

<ResponseField name="data.agent.current_version_id" type="string">
  ID of the current editable draft version.
</ResponseField>

<ResponseField name="data.agent.active_version_id" type="string | null">
  ID of the published version experiments run against, or `null` if never published.
</ResponseField>

<ResponseField name="data.agent.agent_id" type="string | null">
  The voice agent this eval agent is attributed to, or `null` if it isn't scoped to a specific agent.
</ResponseField>

<ResponseField name="data.agent.metadata" type="object">
  Key-value metadata associated with the eval agent. Values are strings.
</ResponseField>

<ResponseField name="data.agent.created_at" type="string">
  ISO 8601 timestamp for when the eval agent was created.
</ResponseField>

<ResponseField name="data.agent.updated_at" type="string">
  ISO 8601 timestamp for when the eval agent was last updated.
</ResponseField>

<ResponseField name="data.agent.deleted_at" type="string | null">
  ISO 8601 timestamp if the eval agent has been soft-deleted, otherwise `null`.
</ResponseField>

<ResponseField name="data.current_version" type="object">
  The current editable draft version of the eval agent.
</ResponseField>

<ResponseField name="data.current_version.id" type="string">
  Unique identifier for this version.
</ResponseField>

<ResponseField name="data.current_version.org_id" type="string">
  ID of the organization that owns this version.
</ResponseField>

<ResponseField name="data.current_version.eval_agent_id" type="string">
  ID of the parent eval agent.
</ResponseField>

<ResponseField name="data.current_version.version_number" type="integer">
  Sequential version number.
</ResponseField>

<ResponseField name="data.current_version.name" type="string">
  Name of this version.
</ResponseField>

<ResponseField name="data.current_version.description" type="string | null">
  Optional description of this version.
</ResponseField>

<ResponseField name="data.current_version.state" type="string">
  State of this version. One of `editable` or `archived`.
</ResponseField>

<ResponseField name="data.current_version.modality" type="string">
  One of `text` or `audio`.
</ResponseField>

<ResponseField name="data.current_version.system_prompt_md" type="string">
  The system prompt for the judge LLM, in Markdown.
</ResponseField>

<ResponseField name="data.current_version.prompt_md" type="string">
  The grading prompt for the judge LLM, in Markdown.
</ResponseField>

<ResponseField name="data.current_version.levels" type="array of objects">
  Verdict levels for graded mode. Empty array for pass/fail mode. Each level contains `level_key`, `label`, `prompt_md`, and optionally `color`.
</ResponseField>

<ResponseField name="data.current_version.target_level_keys" type="array of strings">
  Which level keys count as a target match. Empty array for pass/fail agents.
</ResponseField>

<ResponseField name="data.current_version.weight" type="number">
  Relative weight of this agent in aggregate scoring. Between 0 and 100.
</ResponseField>

<ResponseField name="data.current_version.created_from_version_id" type="string | null">
  ID of the version this was forked from, or `null` if this is the first version.
</ResponseField>

<ResponseField name="data.current_version.created_by" type="string | null">
  Identifier of the user who created this version, or `null`.
</ResponseField>

<ResponseField name="data.current_version.created_at" type="string">
  ISO 8601 timestamp for when this version was created.
</ResponseField>

<ResponseField name="data.current_version.updated_at" type="string">
  ISO 8601 timestamp for when this version was last updated.
</ResponseField>
