---
title: "List Q&A Pairs"
api: "GET https://api.bland.ai/v1/knowledge/{knowledge_base_id}/suggestions"
description: "Returns a paginated list of Q&A pairs (suggestions) for a knowledge base."
---

Lists all Q&A pairs associated with a knowledge base. Q&A pairs are question-and-answer units stored alongside document chunks for direct retrieval.

### Headers

<ParamField header="authorization" type="string" required>
  Your API key for authentication.
</ParamField>

### Path Parameters

<ParamField path="knowledge_base_id" type="string" required>
  The knowledge base ID (UUID).
</ParamField>

### Query Parameters

<ParamField query="type" type="string">
  Filter by type: `"SUGGESTION"` or `"FEEDBACK"`. Returns all types if omitted.
</ParamField>

<ParamField query="page" type="number" default="1">
  Page number for pagination.
</ParamField>

<ParamField query="limit" type="number" default="20">
  Items per page.
</ParamField>

### Response

<ResponseField name="data" type="object">
  Paginated list of Q&A pairs.

  <ResponseField name="data.suggestions" type="array">
    Array of Q&A pair objects.

    <ResponseField name="data.suggestions[].id" type="string">
      Unique identifier for the Q&A pair.
    </ResponseField>

    <ResponseField name="data.suggestions[].knowledge_base_id" type="string">
      Knowledge base this pair belongs to.
    </ResponseField>

    <ResponseField name="data.suggestions[].question" type="string">
      The question.
    </ResponseField>

    <ResponseField name="data.suggestions[].answer" type="string">
      The answer.
    </ResponseField>

    <ResponseField name="data.suggestions[].created_at" type="string">
      ISO timestamp of creation.
    </ResponseField>

    <ResponseField name="data.suggestions[].updated_at" type="string">
      ISO timestamp of last update.
    </ResponseField>
  </ResponseField>

  <ResponseField name="data.total" type="number">
    Total Q&A pairs matching the filter.
  </ResponseField>

  <ResponseField name="data.page" type="number">
    Current page.
  </ResponseField>

  <ResponseField name="data.limit" type="number">
    Items per page.
  </ResponseField>

  <ResponseField name="data.has_more" type="boolean">
    Whether more pages exist.
  </ResponseField>
</ResponseField>

<ResponseField name="errors" type="null">
  Will be `null` on successful request.
</ResponseField>
