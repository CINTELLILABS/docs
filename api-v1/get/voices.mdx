---
title: "List Voice Options"
api: "GET https://api.bland.ai/v1/voices"
description: "Retrieves all available voices for your account, including custom and default presets."
---

### Headers

<ParamField header="authorization" type="string" required>
  Your API key for authentication.
</ParamField>

### Query Parameters

<ParamField query="default_only" type="boolean">
  Filter to show only default presets, excluding user-created presets.
</ParamField>

<ParamField query="presets_only" type="boolean">
  Filter to show only presets.
</ParamField>

<ParamField query="voices_only" type="boolean">
  Filter to show only voices.
</ParamField>

<ParamField query="is_custom" type="boolean">
  Filter for custom voices.
</ParamField>

<ParamField query="name" type="string">
  Filters the `voices` array by the name of the voice.
</ParamField>

<ParamField query="voice_name" type="string">
  Filters the `presets` array by the name of the voice.
</ParamField>

<ParamField query="voice_id" type="integer">
  Filter by the ID of the voice.
</ParamField>

<ParamField query="reduce_latency" type="boolean">
  Filter by latency reduction option. Use in tandem with `reduce_latency` to simulate exactly which voice your agent would use.
</ParamField>

<ParamField query="public" type="boolean">
  Filter to show only public presets, or set to false to exclude them.
</ParamField>

<ParamField query="editable" type="boolean">
  Filter to show only editable presets.
</ParamField>

### Response

<ResponseField type="array" name="voices">
    Contains a list of the voices available for your account, including custom and default presets.
</ResponseField>

<ResponseField type="object" name="presets">
    Contains a list of presets with their details. Each preset object includes `id`, `voice_name`, `description`, `reduce_latency`, `interruption_threshold`, `language`, `voice_settings`, `editable`, and `public`.
</ResponseField>
