---
title: "List Contacts"
api: "GET https://api.bland.ai/v1/contacts"
description: "List all contacts for your organization. Each contact includes identifiers, contact_memories (per persona/agent), and memory entities. Paginated."
---

### Headers

<ParamField header="authorization" type="string" required>
  Your API key for authentication.
</ParamField>

### Query Parameters

<ParamField query="limit" type="integer">
  Maximum number of contacts to return. Default 50, max 200.
</ParamField>

<ParamField query="offset" type="integer">
  Number of contacts to skip (for pagination). Default 0.
</ParamField>

### Response

<ResponseField name="data" type="array">
  Array of contact objects. Each includes identifiers, memories (with entities).
</ResponseField>

<ResponseField name="pagination" type="object">
  Pagination metadata (total, limit, offset, has_more).
</ResponseField>

<ResponseField name="pagination.total" type="integer">
  Total number of contacts in the org.
</ResponseField>

<ResponseField name="pagination.limit" type="integer">
  Limit used for this request.
</ResponseField>

<ResponseField name="pagination.offset" type="integer">
  Offset used for this request.
</ResponseField>

<ResponseField name="pagination.has_more" type="boolean">
  Whether more contacts exist after this page.
</ResponseField>

<ResponseField name="errors" type="null">
  Error array (null on success).
</ResponseField>
