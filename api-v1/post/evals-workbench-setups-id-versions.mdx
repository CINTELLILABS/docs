---
title: "Create Workbench Setup Version"
api: "POST https://api.bland.ai/v1/evals/workbench-setups/{setup_id}/versions"
description: "Fork a new editable draft version of a workbench setup."
---

### Headers

<ParamField header="authorization" type="string" required>
  Your API key for authentication.
</ParamField>

### Path Parameters

<ParamField path="setup_id" type="string" required>
  The ID of the workbench setup to fork a new version from.
</ParamField>

### Body Parameters

All fields are optional.

<ParamField body="name" type="string">
  Display name for the new version. Between 1 and 200 characters. Defaults to the source version's name.
</ParamField>

<ParamField body="description" type="string | null">
  Optional description for the new version.
</ParamField>

<ParamField body="from_version_id" type="string">
  ID of the version to fork from. Defaults to the current draft if not provided.
</ParamField>

### Response

Returns `201` on success.

<ResponseField name="data" type="object">
  The full newly created version object.

  <Expandable title="version fields">
    <ResponseField name="id" type="string">
      Unique identifier for this version.
    </ResponseField>
    <ResponseField name="org_id" type="string">
      ID of the organization that owns this version.
    </ResponseField>
    <ResponseField name="workbench_setup_id" type="string">
      ID of the parent workbench setup.
    </ResponseField>
    <ResponseField name="version_number" type="integer">
      Monotonically increasing version number.
    </ResponseField>
    <ResponseField name="name" type="string">
      Display name of this version.
    </ResponseField>
    <ResponseField name="description" type="string | null">
      Description of this version, or `null` if not set.
    </ResponseField>
    <ResponseField name="state" type="string">
      Always `"editable"` for a newly created version.
    </ResponseField>
    <ResponseField name="attached_agents" type="array">
      Eval agents inherited from the source version. Each item contains `eval_agent_id`, `eval_agent_version_id`, `weight` (0-100), and `target_level_keys` (array of strings).
    </ResponseField>
    <ResponseField name="pass_threshold_pct" type="integer | null">
      Inherited pass threshold percentage (0-100), or `null` if not set.
    </ResponseField>
    <ResponseField name="run_mode" type="string">
      Inherited run mode. One of `text`, `audio`, or `full`.
    </ResponseField>
    <ResponseField name="default_test_config_id" type="string | null">
      Inherited default test configuration ID, or `null` if not set.
    </ResponseField>
    <ResponseField name="default_call_ids" type="array of strings">
      Inherited default call IDs. Up to 5000 entries.
    </ResponseField>
    <ResponseField name="created_from_version_id" type="string | null">
      ID of the version this was forked from.
    </ResponseField>
    <ResponseField name="created_by" type="string | null">
      Identifier of the user who created this version, or `null`.
    </ResponseField>
    <ResponseField name="created_at" type="string">
      ISO 8601 timestamp for when this version was created.
    </ResponseField>
    <ResponseField name="updated_at" type="string">
      ISO 8601 timestamp for when this version was last updated.
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="errors" type="null">
  `null` on success.
</ResponseField>
