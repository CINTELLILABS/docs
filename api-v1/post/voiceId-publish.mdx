---
title: "Publish Cloned Voice"  
api: "POST https://api.bland.ai/v1/voices/{id}/publish"
description: "Make your cloned voice public to use."
---

### Headers

<ParamField header="authorization" type="string" required>
  Your API key for authentication.
</ParamField>

### Path Parameters

<ParamField path="voiceId" type="string" required>
  The ID of the voice to generate the audio sample for, or it's name (like "maya").
</ParamField>

### Response

<ResponseField type="string" name="status">
    The status of the request
</ResponseField>

<ResponseField type="string" name="message">
    Information regarding the status of your request.
</ResponseField>

<ResponseField type="object" name="voice">
    Object containing data of the voice.
</ResponseField>

<ResponseField type="string" name="voice.id">
    Id of the voice.
</ResponseField>

<ResponseField type="string" name="voice.name">
    Name of the voice
</ResponseField>

<ResponseField type="string" name="voice.tags">
    Tags for the voice.
</ResponseField>

<ResponseExample>
```json
{
  "status": "success",
  "message": "Voice published successfully",
  "voice": {
    "id": "fad27b44-667b-4ad7-a9e3-aba1ca69446b",
    "name": "Public - My Example",
    "tags": ["Exciting", "happy"],
    "ratings": 5,
    "description": "Exciting and upbeat voice."
  }
}
```
</ResponseExample>
